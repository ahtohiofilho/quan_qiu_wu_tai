def valor_oferta(quantidades, base=2.0):
    # equação do minério
    # ordena do maior para o menor
    quantidades_ordenadas = sorted(quantidades, reverse=True)
    # soma aplicando a fórmula
    return sum(q * (base ** (p)) for p, q in enumerate(quantidades_ordenadas))
